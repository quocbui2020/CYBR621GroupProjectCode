//------------START IR REMOTE METHODS----------------//
void IRremote()
{
  int cases;
  //If the IR-receiver receives a signal:
  if (IrReceiver.decode())
  {
    if (IrReceiver.decodedIRData.decodedRawData == 0xFFFFFFFF)
    {
      IrReceiver.decodedIRData.decodedRawData = Previous;
    }

    switch (IrReceiver.decodedIRData.decodedRawData)
    {
      case 0xBB44FF00: //Button |<<
        StealWifiPassword();
        Serial.println("Detected button |<<");
        break;
      case 0xBC43FF00:  //Button >>|
        ExportWifiTxtFile();
        Serial.println("Detected button >>|");
        break;
      default: break; //Must be some errors
    }
    Serial.println(IrReceiver.decodedIRData.decodedRawData, HEX);
    IrReceiver.resume();
  }
}

//------------START I2C LCD FUNCTION----------------//
void lcd_setCursor(int column, int row)
{
  char pos = ((0x40) * row) + column;
  pos |= B10000000;
  Wire.beginTransmission(62);
  Wire.write(B00000000);
  delay(1);
  Wire.write(pos);
  Wire.endTransmission(62);
}
void lcd_printStr(char str[])
{
  int index;
  lcd_setCursor(0, 0); //reset the cursor back to pos (0,0)
  Wire.beginTransmission(62);
  Wire.write(B00000001); //Clear The Screen
  while (index == 0)
  {
    //Case: When the char is not the last letter
    if (str[index + 1] != '\0')
    {
      Wire.write(B11000000);  //Send control byte, tell slave "not the last char"
      Wire.write(str[index]); //Send the byte data - letter
    }
    //Case: When the char is the last letter
    else
    {
      Wire.write(B01000000); //Send control byte, tell slave "this is last the char"
      Wire.write(str[index]); //Send the byte data - letter
      Wire.endTransmission(62);
      break;
    }
    index++;
  }
}

//------------PROGRAM START HERE----------------//
void setup() {
  Serial.begin(9600);
  Keyboard.begin();
  //enable the IR remote:
  IrReceiver.begin(IRreceiverPIN, ENABLE_LED_FEEDBACK);

  //I2C LCD Screen Setup:
  Wire.begin();     //Assign Arduino as Master
  Wire.setClock(100000);
  Wire.beginTransmission(62); //Slave address of I2C LCD
  delay(50);
  Wire.write(B00111000);   // function set, normal instruction mode
  Wire.write(B00111001);   // function set, extended instruction mode
  Wire.write(B00010100);   // interval osc
  Wire.write(B01110000);   // contrast C3-C0 //Contrasting setting is C0-C3
  Wire.write(B01011110);   // C5-C4, Ion, Bon ////Contrasting setting isC4-C5
  Wire.write(B01101100);  // follower control
  delay(200);
  Wire.write(B00111000);  // function set, normal instruction mode
  Wire.write(B00001110);  // Display On
  Wire.write(B00000001);  // Clear Display
  Wire.endTransmission();

}

void loop() {
  //Keep checking for the input from the IR remote:
  IRremote();
}
